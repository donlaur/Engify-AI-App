```typescript
// src/app/for-{role}/page.tsx (e.g., /for-engineers/page.tsx)
export default async function ForEngineersPage() {
  const prompts = await getRolePrompts('engineer');
  const patterns = await getRolePatterns('engineer');
  
  return (
    <div>
      <RoleHero role="Engineers" stats={prompts.length} />
      <FeaturedPrompts prompts={prompts.slice(0, 8)} />

}
```

**Option 2: Dynamic Route (Alternative)**
```typescript
// src/app/roles/[role]/page.tsx
export default async function RoleLandingPage({ params }: { params: { role: string } }) {
  // Map URL slug to database role value
  const roleMap: Record<string, string> = {
    'engineers': 'engineer',
    'engineering-managers': 'engineering-manager',
    'product-managers': 'product-manager',
    'qa-engineers': 'qa',
    'architects': 'architect',
    'devops-sre': 'devops-sre',
    'scrum-masters': 'scrum-master',
    'product-owners': 'product-owner',
    'directors': 'director',
    'designers': 'designer',
  };
  
  const dbRole = roleMap[params.role] || params.role;
  const prompts = await getRolePrompts(dbRole);
  const patterns = await getRolePatterns(dbRole);
  const roleInfo = getRoleInfo(dbRole);
  
  return (
    <div>
      <RoleHero role={roleInfo} stats={prompts.length} />
      <FeaturedPrompts prompts={prompts.slice(0, 8)} />
      <PatternShowcase patterns={patterns} />
      <UseCasesSection role={dbRole} />
      <CategoriesSection prompts={prompts} />
      <CTASection />
    </div>
  );
}
```

**Recommendation:** Use Option 1 (static pages) for consistency with existing `/for-*` pattern.
